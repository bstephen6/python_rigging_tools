string $mirrorCtlTarget = "INSERT NAME OF MIRROR CTL TARGET";
string $jointTarget = "INSERT NAME OF MIRRORED JOINT";
string $intendedCtlDupName = "INSERT MIRRORED NAME" ;
vector $targetLocation =`xform -q -ws -t $jointTarget`;

duplicate $mirrorCtlTarget;

string $duplicateCtlCurve = ($mirrorCtlTarget + "1");
select $duplicateCtlCurve ;

parent -w;
move -rpr 0 0 0 ;

setAttr ( $duplicateCtlCurve + ".rotateX" ) 0;
setAttr ( $duplicateCtlCurve + ".rotateY" ) 0;
setAttr ( $duplicateCtlCurve + ".rotateZ" ) 0;

FreezeTransformations;

setAttr ( $duplicateCtlCurve + ".translateX" ) ($targetLocation.x);
setAttr ( $duplicateCtlCurve + ".translateY" ) ($targetLocation.y);
setAttr ( $duplicateCtlCurve + ".translateZ" ) ($targetLocation.z);

rename $duplicateCtlCurve $intendedCtlDupName;

select -cl ;
