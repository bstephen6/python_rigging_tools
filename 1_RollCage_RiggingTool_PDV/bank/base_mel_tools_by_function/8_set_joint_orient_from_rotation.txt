string $joint = "joint";


float $jointRotationX = `getAttr ( $joint + ".rotateX" )`;
float $jointRotationY = `getAttr ( $joint + ".rotateY" )`;
float $jointRotationZ = `getAttr ( $joint + ".rotateZ" )`;

setAttr ( $joint + ".rotateX" ) 0;
setAttr ( $joint + ".rotateY" ) 0;
setAttr ( $joint + ".rotateZ" ) 0;


setAttr ( $joint + ".jointOrientX" ) $jointRotationX;
setAttr ( $joint + ".jointOrientY" ) $jointRotationY;
setAttr ( $joint + ".jointOrientZ" ) $jointRotationZ;
