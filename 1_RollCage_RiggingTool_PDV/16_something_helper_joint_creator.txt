string $helperJointBase = "L_shoulder";
string $helperCtlBase = ($helperJointBase + "_ctl");
string $helperJoint1 = ($helperJointBase + "_helper_1") ;
string $helperJoint2 = ($helperJointBase + "_helper_2") ;

string $helperLoc1 = ($helperJointBase + "_locator_1") ;
vector $helperLoc1Cord =`xform -q -ws -t $helperJointBase`;
string $helperLoc1Group = ($helperLoc1 + "_group") ;

string $helperLoc2 = ($helperJointBase + "_locator_2") ;
vector $helperLoc2Cord =`xform -q -ws -t $helperJoint1`;
string $helperLoc2Group = ($helperLoc2 + "_group") ;

string $helperLoc3 = ($helperJointBase + "_locator_3") ;
vector $helperLoc3Cord =`xform -q -ws -t $helperJoint2`;
string $helperLoc3Group = ($helperLoc3 + "_group") ;


float $helperLocRotX = `getAttr ( $helperCtlBase + "_group.rotateX" )` ;
float $helperLocRotY = `getAttr ( $helperCtlBase + "_group.rotateY" )` ;
float $helperLocRotZ = `getAttr ( $helperCtlBase + "_group.rotateZ" )` ;






CreateLocator;

rename "locator1" $helperLoc1 ;

setAttr ( $helperLoc1 + ".translateX" ) ($helperLoc1Cord.x);
setAttr ( $helperLoc1 + ".translateY" ) ($helperLoc1Cord.y);
setAttr ( $helperLoc1 + ".translateZ" ) ($helperLoc1Cord.z);

select -r $helperLoc1;

doGroup 0 1 1;

rename "group1" $helperLoc1Group ;

select $helperLoc1Group;

SnapToPoint; dR_enterForSnap;
move -rpr ($helperLoc1Cord.x) ($helperLoc1Cord.y) ($helperLoc1Cord.z) ($helperLoc1Group + ".scalePivot") ($helperLoc1Group + ".rotatePivot") ;
SnapToPoint; dR_enterForSnap;

select -cl ;

setAttr ( $helperLoc1Group + ".rotateX") $helperLocRotX ;
setAttr ( $helperLoc1Group + ".rotateY") $helperLocRotY ;
setAttr ( $helperLoc1Group + ".rotateZ") $helperLocRotZ ;











CreateLocator;

rename "locator1" $helperLoc2 ;

setAttr ( $helperLoc2 + ".translateX" ) ($helperLoc2Cord.x);
setAttr ( $helperLoc2 + ".translateY" ) ($helperLoc2Cord.y);
setAttr ( $helperLoc2 + ".translateZ" ) ($helperLoc2Cord.z);

select -r $helperLoc2;

doGroup 0 1 1;

rename "group1" $helperLoc2Group ;

select $helperLoc2Group;

SnapToPoint; dR_enterForSnap;
move -rpr ($helperLoc2Cord.x) ($helperLoc2Cord.y) ($helperLoc2Cord.z) ($helperLoc2Group + ".scalePivot") ($helperLoc2Group + ".rotatePivot") ;
SnapToPoint; dR_enterForSnap;

select -cl ;


setAttr ( $helperLoc2Group + ".rotateX") $helperLocRotX ;
setAttr ( $helperLoc2Group + ".rotateY") $helperLocRotY ;
setAttr ( $helperLoc2Group + ".rotateZ") $helperLocRotZ ;










CreateLocator;

rename "locator1" $helperLoc3 ;

setAttr ( $helperLoc3 + ".translateX" ) ($helperLoc3Cord.x);
setAttr ( $helperLoc3 + ".translateY" ) ($helperLoc3Cord.y);
setAttr ( $helperLoc3 + ".translateZ" ) ($helperLoc3Cord.z);

select -r $helperLoc3;

doGroup 0 1 1;

rename "group1" $helperLoc3Group ;

select $helperLoc3Group;

SnapToPoint; dR_enterForSnap;
move -rpr ($helperLoc3Cord.x) ($helperLoc3Cord.y) ($helperLoc3Cord.z) ($helperLoc3Group + ".scalePivot") ($helperLoc3Group + ".rotatePivot") ;
SnapToPoint; dR_enterForSnap;

select -cl ;

setAttr ( $helperLoc3Group + ".rotateX") $helperLocRotX ;
setAttr ( $helperLoc3Group + ".rotateY") $helperLocRotY ;
setAttr ( $helperLoc3Group + ".rotateZ") $helperLocRotZ ;










parent $helperLoc1Group $helperCtlBase ;

parent $helperLoc2Group $helperCtlBase ;

parent $helperLoc3Group $helperCtlBase ;







select -r $helperCtlBase ;
select -tgl $helperLoc1 ;
doCreateOrientConstraintArgList 1 { "1","0","0","0","1","0","0","1","","1" };
orientConstraint -mo -skip x -weight 1;


select -cl ;



select -r $helperCtlBase ;
select -tgl $helperLoc2 ;
doCreateOrientConstraintArgList 1 { "1","0","0","0","1","0","0","1","","1" };
orientConstraint -mo -skip x -weight 1;

select -cl ;


select -r $helperCtlBase ;
select -tgl $helperLoc3 ;
doCreateOrientConstraintArgList 1 { "1","0","0","0","1","0","0","1","","1" };
orientConstraint -mo -skip x -weight 1;


select -cl ;



select -r $helperLoc1 ;
select -tgl $helperJointBase ;
doCreateOrientConstraintArgList 1 { "1","0","0","0","0","0","0","1","","1" };
orientConstraint -mo -weight 1;


select -cl ;


select -r $helperLoc2 ;
select -tgl $helperJoint1 ;
doCreateOrientConstraintArgList 1 { "1","0","0","0","0","0","0","1","","1" };
orientConstraint -mo -weight 1;


select -cl ;


select -r $helperLoc3 ;
select -tgl $helperJoint2 ;
doCreateOrientConstraintArgList 1 { "1","0","0","0","0","0","0","1","","1" };
orientConstraint -mo -weight 1;


select -cl ;


setAttr ($helperLoc1 + ".visibility") 0;
setAttr ($helperLoc2 + ".visibility") 0;
setAttr ($helperLoc3 + ".visibility") 0;
